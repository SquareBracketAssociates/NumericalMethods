{\bf lowerRows:} {\tt anArrayOfArray}
\begin{verbatim}
    | n v |
    n := anArrayOfArray size.
    rows := DhbVector new: n.
    1 to: n do:
        [ :i |
          v := DhbVector new: n.
          1 to: i do: [ :j | v at: j put: ( ( anArrayOfArray at: i) 
                                                              at: j)].
          ( i + 1) to: n do: [ :j | v at: j put: ( ( anArrayOfArray 
                                                       at: j) at: i)].
          rows at: i put: v.
        ].
    ^self

\end{verbatim}
