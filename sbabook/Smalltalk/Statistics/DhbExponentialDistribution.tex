{\bf fromHistogram:} {\tt aHistogram}
\begin{verbatim}
    | mu |
    aHistogram minimum < 0
        ifTrue: [ ^nil].
    mu := aHistogram average.
    ^mu > 0 ifTrue: [ self scale: aHistogram average]
            ifFalse:[ nil]

\end{verbatim}
